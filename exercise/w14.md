## [442 - 向量加法](https://neoj.sprout.tw/problem/442/)
```cpp
#include<iostream>
#include<sstream>
#include<vector>
using namespace std;
int main(){
    string s1,s2;
    getline(cin,s1);
    getline(cin,s2);
    
    stringstream ss1;
    stringstream ss2;

    int val1,val2;
    ss1 << s1;
    ss2 << s2;

    vector<int> ans;
    while(ss1 >> val1 && ss2 >> val2){
        ans.push_back(val1+val2);
    }
    for(int i=0;i < ans.size();i++){
        cout << ans[i];
        if(i!=ans.size()-1) cout << ' ';
    }
    cout << endl;
}
```
